    function AssetsWebpackPlugin(options) {
  this.options = merge({}, {
    path: '.',
    filename: 'webpack-assets.json',
    prettyPrint: false,
    update: false,
    fullPath: true
  }, options)
  this.writer = createQueuedWriter(createOutputWriter(this.options))
}
